## Critical implementation facts (verified before plan writing)

1. **`/pof/status` is reached two ways.** Either via the Next.js proxy `GET /api/pof-bridge/status?port=30040` (which fetches `http://127.0.0.1:30040/pof/status` server-side) OR via direct client fetches in `pofBridgeConnection` (manager handles direct `http://127.0.0.1:30040/pof/status`). Playwright's `page.route('**/pof/status', ...)` matches the URL path regardless of host, catching both.
2. **`BuildProfile` shape** is at `src/lib/packaging/build-profiles.ts:45-65`. Use `createDefaultProfile('Win64', 'Shipping')` from that file as a starting point; the POST endpoint generates `id`/`createdAt`/`updatedAt`.
3. **CompactTerminal listens at `window` bubble phase** (`src/components/cli/CompactTerminal.tsx:87`, no capture flag). The harness's capture-phase listener fires first; `stopImmediatePropagation()` prevents the terminal handler.
4. **Zustand stores are NOT exposed on `window`.** Live-wait helpers must use DOM polling (`pof-cli-panel-running-indicator` testId for CLI; `pof-cook-progress-result` for cook).
5. **`writeFindings()` currently encodes filename as `{date}-{mode}.md`.** For D2 we want `{date}-live-d2.md` not `{date}-live.md` to avoid colliding with future D3 runs. Need a filename suffix override.

```typescript
export interface WaitResult {
  success: boolean;
  durationMs: number;
  timedOut: boolean;
  /** Captured terminal output excerpt (live mode only — empty/null in stub). */
  outputExcerpt?: string;
}

/**
 * Wait until a CLI session transitions running → stopped, OR timeoutMs elapses.
 *
 * Polling strategy: DOM-based via the `pof-cli-panel-running-indicator`
 * testId (added by sub-project B). When the indicator is no longer attached,
 * the active session is considered done.
 *
 * Stub mode: short-circuits with success=true after ~200ms (no real session
 * runs because capture-phase listener stopped propagation).
 *
 * On timeout: tries to click a Stop button (if visible) to abort; records
 * timedOut: true and success: false.
 */
export async function waitForCliComplete(
  page: Page,
  sessionLabel: string,
  timeoutMs: number = 600_000,
): Promise<WaitResult> {
  const start = Date.now();

  // Stub mode short-circuit. The capture-phase listener stops propagation
  // so no real session ever starts; isRunning never flips true.
  if (process.env.HARNESS_MODE !== 'live') {
    await page.waitForTimeout(200);
    return { success: true, durationMs: Date.now() - start, timedOut: false };
  }

  // Live mode: wait for the running indicator to appear (proves the session started),
  // then wait for it to disappear (session finished). If it never appears within
  // a short grace window, the dispatch likely never fired — treat as failure.
  const indicator = page.getByTestId('pof-cli-panel-running-indicator');
  try {
    await indicator.first().waitFor({ state: 'attached', timeout: 10_000 });
  } catch {
    return {
      success: false,
      durationMs: Date.now() - start,
      timedOut: false,
      outputExcerpt: `waitForCliComplete(${sessionLabel}): running indicator never appeared within 10s — dispatch likely never fired`,
    };
  }

  // Now wait for the indicator to disappear (session done) OR timeoutMs.
  try {
    await indicator.first().waitFor({ state: 'detached', timeout: timeoutMs });
  } catch {
    // Timeout — try to capture output before aborting.
    let output = '';
    try {
      const outputEl = page.getByTestId('pof-cli-panel-output');
      output = (await outputEl.first().textContent({ timeout: 2000 })) ?? '';
    } catch { /* noop */ }

    // Try to click a Stop button if one is visible.
    try {
      const stopBtn = page.locator('button[aria-label*="abort" i], button[aria-label*="stop" i]').first();
      if ((await stopBtn.count()) > 0) {
        await stopBtn.click({ timeout: 2000 });
      }
    } catch { /* noop — best-effort abort */ }

    return {
      success: false,
      durationMs: Date.now() - start,
      timedOut: true,
      outputExcerpt: output.slice(-2000),
    };
  }

  // Success — capture the tail of the output for the findings doc.
  let output = '';
  try {
    const outputEl = page.getByTestId('pof-cli-panel-output');
    output = (await outputEl.first().textContent({ timeout: 2000 })) ?? '';
  } catch { /* noop */ }

  return {
    success: true,
    durationMs: Date.now() - start,
    timedOut: false,
    outputExcerpt: output.slice(-2000),
  };
}

/**
 * Wait until the cook completes, signalled by the appearance of the
 * pof-cook-progress-result testId (added by sub-project B).
 *
 * Stub mode: synthetic SSE in setupHarnessMode finishes in <2s; this
 * just waits for the result element.
 */
export async function waitForCookComplete(
  page: Page,
  timeoutMs: number = 600_000,
): Promise<WaitResult> {
  const start = Date.now();
  const result = page.getByTestId('pof-cook-progress-result');
  try {
    await result.first().waitFor({ state: 'visible', timeout: timeoutMs });
    const status = await result.first().getAttribute('data-status');
    return {
      success: status === 'success',
      durationMs: Date.now() - start,
      timedOut: false,
      outputExcerpt: (await result.first().textContent()) ?? '',
    };
  } catch {
    return { success: false, durationMs: Date.now() - start, timedOut: true };
  }
}
```

  harness: HarnessHandle,
  page: Page,
  label: string,
  body: () => Promise<{ success: boolean; durationMs: number; timedOut: boolean; notes?: string }>,

      step: label,
      status: result.timedOut ? 'fail' : (result.success ? 'pass' : 'fail'),
      durationMs: result.durationMs,
      notes: result.notes,
    });
  });
